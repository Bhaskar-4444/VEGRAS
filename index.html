    function showSlide(index) {
        slides.forEach((slide, i) => {
            slide.classList.remove('active', 'prev');
            if (i === index) {
                slide.classList.add('active');
            } else if (i === (index - 1 + slides.length) % slides.length) {
                slide.classList.add('prev');
            }
        });
        navDots.forEach((dot, i) => {
            dot.classList.toggle('active', i === index);
        });
    }

    function nextSlide() {
        currentSlide = (currentSlide + 1) % slides.length;
        showSlide(currentSlide);
    }

    // Auto-slide every 5 seconds
    setInterval(nextSlide, 5000);

    // Navigation dots
    navDots.forEach((dot, index) => {
        dot.addEventListener('click', () => {
            currentSlide = index;
            showSlide(currentSlide);
        });
    });

        rects.forEach((rect, index) => {

                    rect.classList.add('animate');
                    texts.forEach((text, tIndex) => {
                        if (tIndex <= index * 2 + 1) { // Approximate for suppliers
                            setTimeout(() => text.classList.add('animate'), 200);
                        }
                    });
                }, index * 300);
            }
        });

        }, 900);

        // Animate green paths
        paths.forEach((path, index) => {
            if (index < 3) { // Green paths
                setTimeout(() => path.classList.add('animate'), 1200 + index * 200);
            }
        });

        // Animate customer rects

            document.querySelectorAll('.flow-rect').forEach((rect, index) => {

                if (tIndex > 8) { // Approximate for customers
